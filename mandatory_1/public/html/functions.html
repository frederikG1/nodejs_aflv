1. Greet Function

A function that takes a name as a parameter and returns a greeting.


            function greet(name) {
                return "Hello, " + name;
            }
            console.log(greet("Frederik"));
Output: Hello, Frederik

2. Random Integer Function

A function that generates a random int between a range (min and max).


            function getRandomInt(min, max) {
                return Math.floor(Math.random() * (max + 1 - min) + min);
            }
            console.log(getRandomInt(4, 9));
Output: Random integer between 4 and 9

3. Callback Function

Callback function is a function that takes another function (=>).


            function doTwice(action) {
                action();
                action();
            }
            doTwice(() => console.log("Working"));
Output: Working
Working

4. Perform Action Function

Another example of a callback function using const.


            function performAction(name, action) {
                return action(name);
            }

            const workingAction = (name) => {
                return `${name} is working`;
            }
            console.log(workingAction("Frederik", workingAction));
Output: Frederik is working
